import os
from typing import Dict
from pydantic import BaseModel, Field
from pathlib import Path

from agno.agent import Agent
from agno.media import Image
from agno.models.vllm import VLLM
from pydantic import BaseModel, Field
from rich.pretty import pprint

# === 1) Схема результата ===
class StampInfo(BaseModel):
    # ВАЖНО: это именно материал, а не организация
    материал: str = Field(
        "",
        description="Материал (как указан в штампе/поле материала/ показывай всё значение этого поля), например: 'Сталь Р650HS Schoeller-Bleckmann'"
    )
    номер_чертежа: str = Field("", description="Номер чертежа/документа")
    название_чертежа: str = Field("", description="Название чертежа/изделия")
    разработчик: str = Field("", description="ФИО/подпись разработчика (если есть)")
    дата: str = Field(
        "",
        description="Одна общая дата из штампа (самая явная/основная). Если дат несколько — возьми наиболее основную/крупно указанную."
    )
    дополнительные_поля: Dict[str, str] = Field(
        default_factory=dict,
        description=(
            "Прочие поля штампа в формате 'название_поля: значение'. "
            "Примеры: Лист, Листов, Формат, Лит., Масса, Масштаб, Изм., № докум., Подп., Пров., "
            "Т.контр., Н.контр., Утв. Если значение не читается/пустое — ''."
        ),
    )
	
# === 2) Настройка окружения/модели ===
os.environ["VLLM_API_KEY"] = "local"

# Предполагается, что у тебя уже есть эти импорты в проекте:
# from some_agent_lib import Agent, VLLM, Image
# Ниже — как в твоём примере:
agent = Agent(
    model=VLLM(id="qwen3-vl", base_url="http://172.18.12.45:8099/v1"),
    output_schema=StampInfo,
    description="Агент для анализа чертежей и распознавания штампов (выводит значения полей).",
)

current_dir = Path.cwd()
image_path = current_dir / "ramka4.jpg"

# === 3) Промт: заставляем вернуть пары ключ=значение, а не только названия ===
prompt = """
Ты OCR-агент. Считай текст именно из штампа чертежа на изображении.

Верни результат СТРОГО по output_schema (StampInfo).

Правила заполнения:
1) Заполняй значения, как они написаны на изображении. Не пиши названия полей вместо значений.
2) Поле `материал`: это именно материал (например, "Сталь ..."), НЕ название организации.
3) Поле `дата`: верни ОДНУ общую дату из штампа (самую основную/явную). Если дат несколько — выбери наиболее основную.
4) Поле `дополнительные_поля`: заполняй как пары "название_поля": "значение".
   Включай поля типа: "Лист", "Листов", "Формат", "Лит.", "Масса", "Масштаб",
   "Изм.", "№ докум.", "Подп.", "Пров.", "Т.контр.", "Н.контр.", "Утв." и т.п.
5) Если значение пустое или не читается — поставь пустую строку "".
6) Сохраняй язык и формат как в оригинале (точки, запятые, 1:2, 6,3, A4×4).
"""

# === 4) Запуск ===
# image_path должен быть определён заранее
response = agent.run(
    prompt,
    images=[Image(filepath=image_path)],
    stream=True,
)

# === 5) Печать результата ===
from pprint import pprint

for event in response:
    pprint(event.content)
